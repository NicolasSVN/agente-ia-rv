SE for (a) — JSON bruto — implemente a seguinte correção:

Antes de chamar _generate_embedding() para blocos do tipo TABLE,
aplique o semantic_transformer:

from services.semantic_transformer import SemanticTransformer

def _prepare_content_for_embedding(self, block_content: str,
                                    block_type: str) -> str:
    if block_type == "TABLE":
        try:
            table_data = json.loads(block_content)
            transformer = SemanticTransformer()
            semantic = transformer.parse_table_to_semantic(table_data)
            narrative = transformer.semantic_to_narrative_chunks(semantic)
            # Usa narrativa se gerada com sucesso e tem conteúdo
            if narrative and len(narrative) > len(block_content) * 0.3:
                return narrative
        except Exception:
            pass  # Fallback para o conteúdo original se falhar
    return block_content

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

No document_processor.py, torne o DPI adaptivo ao tipo de
conteúdo detectado. Como o Vision classifica o tipo ANTES
de processar, use uma abordagem em dois passes para páginas
com tabelas densas:

DPI_BY_CONTENT = {
    "text": 150,        # Suficiente para texto corrido
    "mixed": 200,       # Mais seguro para combinações
    "table": 250,       # Tabelas densas precisam de mais detalhe
    "infographic": 200, # Gráficos se beneficiam de mais resolução
    "image_only": 150,  # Sem texto, DPI não ajuda
}

# Primeiro passe: 150 DPI para classificar o tipo da página
# Se for 'table' ou 'mixed', re-processa a 250 DPI

Implemente como uma função separada para não afetar o
pipeline atual em caso de erro:

def _get_optimal_dpi(self, content_type: str) -> int:
    return DPI_BY_CONTENT.get(content_type, 150)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

No upload_validator.py, após gerar o file_hash, verifique
se ele já existe no banco:

from database.models import Material  # ou onde file_hash é armazenado

def check_duplicate_file(self, file_hash: str,
                          db: Session) -> Optional[dict]:
    existing = db.query(Material).filter(
        Material.file_hash == file_hash
    ).first()
    if existing:
        return {
            "is_duplicate": True,
            "existing_material_id": existing.id,
            "existing_material_name": existing.name,
            "uploaded_at": existing.created_at.isoformat()
        }
    return None

SELECT
    block_type,
    publish_status,
    count(*) as total,
    count(CASE WHEN content LIKE '{%' THEN 1 END) as json_format,

GROUP BY block_type, publish_status
ORDER BY total DESC;
