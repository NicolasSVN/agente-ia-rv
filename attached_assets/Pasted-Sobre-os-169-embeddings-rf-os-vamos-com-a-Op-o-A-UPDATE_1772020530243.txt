Confirme que o mapeamento entre doc_id e produto está correto.
O doc_id tem formato "product_block_{block_id}". Precisamos garantir
que o bloco referenciado pertence ao produto certo antes de atualizar.

Execute:
SELECT
    de.id,
    de.doc_id,
    de.product_ticker as ticker_errado,
    de.title,
    cb.id as block_id,
    m.name as material_nome,
    p.ticker as ticker_correto,
    p.name as produto_correto
FROM document_embeddings de
JOIN content_blocks cb ON de.doc_id = 'product_block_' || cb.id::text
JOIN materials m ON cb.material_id = m.id
JOIN products p ON m.product_id = p.id
WHERE de.product_ticker = '__SYSTEM_UNASSIGNED__'
LIMIT 20;

ETAPA 2 — UPDATE SQL (só execute após validar Etapa 1):

UPDATE document_embeddings de
SET
    product_ticker = p.ticker,
    product_name   = p.name,
    gestora        = COALESCE(p.gestora, de.gestora),
    category       = COALESCE(p.category, de.category)
FROM content_blocks cb
JOIN materials m ON cb.material_id = m.id
JOIN products p ON m.product_id = p.id
WHERE de.doc_id = 'product_block_' || cb.id::text
  AND de.product_ticker = '__SYSTEM_UNASSIGNED__';
