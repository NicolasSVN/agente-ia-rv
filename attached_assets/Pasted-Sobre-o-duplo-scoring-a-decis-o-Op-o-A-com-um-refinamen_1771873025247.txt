def _calculate_recency_score(self, metadata: dict) -> float:
    """Score de recência baseado em created_at_source ou valid_until."""
    from datetime import datetime, timezone
    
    date_str = metadata.get("created_at_source") or metadata.get("valid_until")
    if not date_str:
        return 0.5  # neutro se sem data
    
    try:
        # Tenta parsear a data (adapte ao formato real do campo)
        doc_date = datetime.fromisoformat(date_str.replace("Z", "+00:00"))
        now = datetime.now(timezone.utc)
        days_old = (now - doc_date).days
        
        # Score decai com idade: doc de hoje = 1.0, doc de 2 anos = ~0.3
        recency_score = max(0.2, 1.0 - (days_old / 730))
        return recency_score
    except:
        return 0.5
